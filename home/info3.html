# Better-scroll是什么

介绍 BetterScroll是一款重点解决移动端各中滚动场景需求的开源插件，适用于滚动列表、选择器、轮播图、索引列表、开屏引导等应用场景

基于原生JavaScript实现，不依赖任何框架，所以既可以原生JavaScript引用，也可以与目前前端MVVM框架结合使用.

<!DOCTYPE html>
<html>
  <head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title></title>
    <style>
      body{
        margin: 0;
      }
      body, html{
        height: 100%;
      }
      ul,ol{
        list-style: none;
        padding: 0;
        margin: 0;
      }
      .con{
        width: 100vw;
        height: 100vh;
        overflow: hidden;
      }
      ul{
        position: relative;
      }
      ul>li{
        width: 100%;
        height: 40px;
        line-height: 40px;
        text-indent: 20px;
        border-bottom: 1px solid #666;
        color: #333;
      }
      .pullUp:after{
        content: "加载中，请稍后。。。";
      }
      .pullDown:before{
        content: "正在刷新，请稍后。。。";
        position: absolute;
        top: -30px;
      }
    </style>
    <script src="./bscroll.js"></script>
  </head>
  <body>
    <div class="con">
      <ul class="list"></ul>
    </div>
    <script>
      var list = document.querySelector('.list');
      function getData() {
        var html = "";
        for (var i = 0; i < 30; i++) {
          html += "<li>我是第"+(i+1)+"个li</li>"
        }
        list.innerHTML = html;
      }
      
      // 初始化con
      function con() {
        var con = document.querySelector(".con");
        var bscroll = new BScroll(con, {
          scrollY: true,
          // 缓冲动画 默认为true
          momentun: true,
          // 上滑加载
          pullUpLoad: {
            // 离底部还有多少距离 px
            threshold: 50
          },
          // 下拉刷新
          pullDownRefresh: {
            // 离底部还有多少距离 px
            threshold: -50,
            // 停留的位置 距离顶部多少px
            stop: 30
          }
        })
        // 上滑加载
        bscroll.on('pullingUp', () => {
          setTimeout(() => {
            getNewData('up');
            list.classList.add('pullUp');
            // 告诉bscroll已经完成了上滑操作
            bscroll.finishPullUp();
            // 刷新dom
            bscroll.refresh();
          }, 1000)
        })
        // 下拉刷新
        bscroll.on('pullingDown', () => {
          list.classList.add("pullDown");
          setTimeout(()=> {
            getNewData('down');
            // 告诉bscroll已经完成了上滑操作
            bscroll.finishPullDown();
            list.classList.remove('pullDown');
            // 刷新dom
            // bscroll.refresh()
          }, 1000)
        })
        
      }
      // 上滑加载 下拉刷新 数据刷新
      function getNewData(type) {
        var html = "";
        for(var i = 0; i < 10; i++) {
          html +="<li>我是新加载的"+(i+1)+"个li</li>"
        }
        if(type === "up") {
          list.innerHTML += html;
        } else {
          list.innerHTML = html + list.innerHTML;
        }
      }
      window.onload = () => {
        getData();
        con();
      }
    </script>
  </body>
</html>

(function() {
  // 获取外框
  var list = document.querySelector('.list-wrapper');
  // 获取导航的容器
  var indexListNav = document.querySelector('.index-list-nav');
  // 获取导航下的li
  var indexListNavs = indexListNav.querySelectorAll('li');
  // 获取顶部索引
  var indexListFixed = document.querySelector('.index-list-fixed');
  
  var indexListContent = document.querySelector('.index-list-content');
  // 获取ul下的所有li
  var indexLists = indexListContent.children[1].children;
  
  var indexList = new BScroll(list, {
    // 事件派发
    probeType: 3
  })
  indexList.on("scroll", (e) => {
    // x y
    // console.log(e);
    var y = -e.y;
    // 未超过第一项就不显示顶部
    if(y < indexLists[0].offsetTop) {
      setNav(0)
      indexListFixed.style.display = 'none';
      return;
    }
    // 顶部城市显示
    indexListFixed.style.display = 'block';
    // 最后一项 i+1 表达式不成立
    for(var i = 0; i < indexLists.length - 1; i++) {
      // 滚动距离是否大于当前项目，并且小于下一项
      if (y >= indexLists[i].offsetTop && 
        y < indexLists[i+1].offsetTop) {
          setNav(i);
          indexListFixed.innerHTML = indexLists[i].children[0].innerHTML;
          return;
        }
    }
    // 处理最后一项
    setNav(indexLists.length - 1);
    indexListFixed.innerHTML = indexLists[indexLists.length - 1].children[0].innerHTML;
  })
  indexListNav.addEventListener('touchstart', (e) => {
    // console.log(e.changedTouches[0].clientY);
    setIndex(e.changedTouches[0].clientY)
  })
  indexListNav.addEventListener('touchmove', (e) => {
    // console.log(e.changedTouches[0].clientY);
    setIndex(e.changedTouches[0].clientY)
  })
  // 设置坐标
  function setIndex(y) {
    var index = getIndex(y);
    // 没有对应元素
    if (index < 0 || index > 8) {
      return;
    }
    // 滚动到对应的元素上面去 过渡的时间
    indexList.scrollToElement(indexLists[index], 100);
  }
  // 拿到坐标
  function getIndex(y) {
    // 获取nav相对于可视去的距离
    var navTop = indexListNav.getBoundingClientRect().top;
    var h = 18; // nav li的高
    // 换算成相对于nav的距离
    var index = parseInt((y - navTop)/ 18); // 0 - 8
    // console.log(index);
    return index;
  }
  // nav active控制
  function setNav(index) {
    // 先获取所有的li标签
    indexListNavs.forEach((li) => {
      li.classList.remove('active')
    })
    indexListNavs[index].classList.add("active")
  }
})();
